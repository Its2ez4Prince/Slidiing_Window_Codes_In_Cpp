#include <bits/stdc++.h>

using namespace std;

int main()
{
    int i=0;
    int j=0;
    string str;
    cin>> str;
    int k;
    cin>> k;
    map<char,int>m;
    int mx = INT_MIN;
    while(j<str.length()){
        m[str[j]]++;
        if(m.size()< k){
            j++;
        }
        else if(m.size()== k ){
            mx= max(mx ,j-i+1);
            j++;
        }
        if(m.size()>k ){
            while(m.size()> k){
                m[str[i]]--;
                if(m[str[i]]==0){
                    m.erase(str[i]);
                }
                i++;
            }
            j++;
        }
    }
    cout<< mx<< endl;

    return 0;
}
